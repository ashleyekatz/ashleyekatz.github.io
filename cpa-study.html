    <script>

        // CPA Study Data Array

        // Add as many terms as you need here!

        const cpaData = [

            {

                term: "Accrual Basis Accounting",

                definition: "Recognizing revenues when earned and expenses when incurred, regardless of when cash changes hands."

            },

            {

                term: "Amortization",

                definition: "The systematic allocation of the cost of an intangible asset over its useful life."

            },

            {

                term: "Goodwill",

                definition: "An intangible asset that arises when a buyer acquires an existing business for a price higher than the fair market value of its net assets."

            },

            {

                term: "EBITDA",

                definition: "Earnings Before Interest, Taxes, Depreciation, and Amortization. A proxy for a company's current operating profitability."

            },

            {

                term: "Contingent Liability",

                definition: "A potential liability that may occur depending on the outcome of an uncertain future event (e.g., pending lawsuits)."

            }

        ];


        // State variables

        let currentIndex = 0;

        let isFlipped = false;


        // DOM Elements

        const flashcard = document.getElementById('flashcard');

        const flashcardContainer = document.getElementById('flashcard-container');

        const termDisplay = document.getElementById('term-display');

        const definitionDisplay = document.getElementById('definition-display');

        const prevBtn = document.getElementById('prev-btn');

        const nextBtn = document.getElementById('next-btn');

        const cardCounter = document.getElementById('card-counter');


        // Initialize the app

        function loadCard(index) {

            // Unflip if it's currently showing the back

            if (isFlipped) {

                flashcard.classList.remove('is-flipped');

                isFlipped = false;

                

                // Wait for the flip animation to finish before changing text

                setTimeout(() => {

                    updateText(index);

                }, 300); // Half of the CSS transition time

            } else {

                updateText(index);

            }

        }


        function updateText(index) {

            termDisplay.textContent = cpaData[index].term;

            definitionDisplay.textContent = cpaData[index].definition;

            cardCounter.textContent = `${index + 1} / ${cpaData.length}`;

            

            // Update button states

            prevBtn.disabled = index === 0;

            nextBtn.disabled = index === cpaData.length - 1;

        }


        // Event Listeners

        flashcardContainer.addEventListener('click', () => {

            flashcard.classList.toggle('is-flipped');

            isFlipped = !isFlipped;

        });


        prevBtn.addEventListener('click', (e) => {

            e.stopPropagation(); // Prevent card from flipping when clicking buttons

            if (currentIndex > 0) {

                currentIndex--;

                loadCard(currentIndex);

            }

        });


        nextBtn.addEventListener('click', (e) => {

            e.stopPropagation(); // Prevent card from flipping when clicking buttons

            if (currentIndex < cpaData.length - 1) {

                currentIndex++;

                loadCard(currentIndex);

            }

        });


        // Add keyboard support (Left/Right arrows for nav, Space/Up/Down to flip)

        document.addEventListener('keydown', (e) => {

            if (e.key === 'ArrowRight' && !nextBtn.disabled) {

                currentIndex++;

                loadCard(currentIndex);

            } else if (e.key === 'ArrowLeft' && !prevBtn.disabled) {

                currentIndex--;

                loadCard(currentIndex);

            } else if (e.key === ' ' || e.key === 'ArrowUp' || e.key === 'ArrowDown') {

                flashcard.classList.toggle('is-flipped');

                isFlipped = !isFlipped;

            }

        });


        // Load the first card on startup

        loadCard(currentIndex);

    </script>
