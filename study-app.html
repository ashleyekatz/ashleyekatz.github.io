    <script>

        const cards = [

            { t: "Revenue Recognition", d: "The principle that revenue is recorded when it is earned, not necessarily when cash is received." },

            { t: "Materiality", d: "The threshold above which missing or incorrect information is considered to have an impact on the decision-making of users." },

            { t: "Going Concern", d: "The assumption that a company will remain in business indefinitely and not be forced to liquidate." },

            { t: "Conservatism", d: "The concept of recognizing expenses and liabilities as soon as possible, but assets and revenues only when they are assured." },

            { t: "Internal Control", d: "A process designed to provide reasonable assurance regarding the achievement of objectives in effectiveness and efficiency." }

        ];


        let index = 0;

        const cardEl = document.getElementById('card');

        const termEl = document.getElementById('term');

        const defEl = document.getElementById('definition');

        const countEl = document.getElementById('counter');


        function updateUI() {

            // Reset flip state before changing content

            cardEl.classList.remove('flipped');

            

            setTimeout(() => {

                termEl.innerText = cards[index].t;

                defEl.innerText = cards[index].d;

                countEl.innerText = `${index + 1} / ${cards.length}`;

                document.getElementById('prevBtn').disabled = index === 0;

                document.getElementById('nextBtn').disabled = index === cards.length - 1;

            }, 150); 

        }


        function flipCard() { cardEl.classList.toggle('flipped'); }

        function nextCard() { if(index < cards.length - 1) { index++; updateUI(); } }

        function prevCard() { if(index > 0) { index--; updateUI(); } }


        // Keyboard Support

        document.addEventListener('keydown', (e) => {

            if(e.code === 'Space') flipCard();

            if(e.code === 'ArrowRight') nextCard();

            if(e.code === 'ArrowLeft') prevCard();

        });


        // Start app

        updateUI();

    </script>
